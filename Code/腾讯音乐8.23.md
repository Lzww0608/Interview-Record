#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

const int MOD = 1e9 + 7;

int numFillingArrays(vector<int>& a, int k) {
	int n = a.size();
	vector<vector<int>> dp(n, vector<int>(k + 1, 0));

	for (int j = 1; j <= k; j++) {
		if (a[0] == 0 || a[0] == j) {
			dp[0][j] = 1;
		}
	}

	for (int i = 1; i < n; i++) {
		for (int j = 1; j <= k; j++) {
			if (a[i] == 0 || a[i] == j) {
				for (int l = 1; l <= j; l++) {
					dp[i][j] = (dp[i][j] + dp[i - 1][l]) % MOD;
				}
			}
		}
	}

	int result = 0;
	for (int j = 1; j <= k; j++) {
		result = (result + dp[n - 1][j]) % MOD;
	}

	return result;
}

int main() {
	int n, k;
	cin >> n >> k;
	vector<int> a(n);

	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}

	cout << numFillingArrays(a, k) << endl;
	return 0;
}
